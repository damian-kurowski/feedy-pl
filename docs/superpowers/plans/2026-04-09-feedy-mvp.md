# Feedy MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a SaaS feed manager that parses XML product feeds from e-commerce stores (Shoper-first), lets users map fields, and generates output feeds for Ceneo/GMC.

**Architecture:** Python FastAPI backend with Celery+Redis for async XML processing, PostgreSQL database, Vue 3 + Tailwind frontend SPA. Monolith deployed on a single VPS behind Nginx.

**Tech Stack:** Python 3.12, FastAPI, SQLAlchemy 2.0, Alembic, Celery, Redis, PostgreSQL 16, Vue 3 (Composition API), Vite, Tailwind CSS, Pinia, Axios, pytest, Vitest.

# backend/alembic/env.py
from logging.config import fileConfig

from alembic import context
from sqlalchemy import engine_from_config, pool

from app.config import settings
from app.database import Base
from app.models import *  # noqa: F401, F403 — register all models

config = context.config
config.set_main_option("sqlalchemy.url", settings.database_url_sync)

if config.config_file_name is not None:
    fileConfig(config.config_file_name)

target_metadata = Base.metadata


def run_migrations_online() -> None:
    connectable = engine_from_config(
        config.get_section(config.config_ini_section, {}),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    with connectable.connect() as connection:
        context.configure(
            connection=connection,
            target_metadata=target_metadata,
            include_schemas=True,
            version_table_schema="public",
        )
        with context.begin_transaction():
            context.run_migrations()


run_migrations_online()

from datetime import datetime, timedelta, timezone

from jose import jwt, JWTError
from passlib.context import CryptContext

from app.config import settings

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")


def hash_password(password: str) -> str:
    return pwd_context.hash(password)


def verify_password(plain: str, hashed: str) -> bool:
    return pwd_context.verify(plain, hashed)


def create_access_token(user_id: int, expires_minutes: int | None = None) -> str:
    expire = datetime.now(timezone.utc) + timedelta(
        minutes=expires_minutes or settings.access_token_expire_minutes
    )
    return jwt.encode({"sub": user_id, "exp": expire}, settings.secret_key, algorithm="HS256")


def create_refresh_token(user_id: int) -> str:
    expire = datetime.now(timezone.utc) + timedelta(days=settings.refresh_token_expire_days)
    return jwt.encode({"sub": user_id, "exp": expire, "type": "refresh"}, settings.secret_key, algorithm="HS256")


def decode_token(token: str) -> dict | None:
    try:
        payload = jwt.decode(token, settings.secret_key, algorithms=["HS256"])
        return payload
    except JWTError:
        return None

from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy.orm import selectinload

from app.database import get_db
from app.deps import get_current_user

from app.services.auth_service import hash_password, verify_password, create_access_token, create_refresh_token

router = APIRouter(prefix="/api/auth", tags=["auth"])


@router.post("/register", response_model=TokenResponse, status_code=status.HTTP_201_CREATED)
async def register(body: RegisterRequest, db: AsyncSession = Depends(get_db)):
    existing = await db.execute(select(User).where(User.email == body.email))
    if existing.scalar_one_or_none():
        raise HTTPException(status_code=400, detail="Email already registered")

    user = User(email=body.email, password_hash=hash_password(body.password), plan_id=1)
    db.add(user)
    await db.commit()
    await db.refresh(user)

    return TokenResponse(
        access_token=create_access_token(user.id),
        refresh_token=create_refresh_token(user.id),
    )


@router.post("/login", response_model=TokenResponse)
async def login(body: LoginRequest, db: AsyncSession = Depends(get_db)):
    result = await db.execute(select(User).where(User.email == body.email))
    user = result.scalar_one_or_none()
    if not user or not verify_password(body.password, user.password_hash):
        raise HTTPException(status_code=401, detail="Invalid credentials")

    return TokenResponse(
        access_token=create_access_token(user.id),
        refresh_token=create_refresh_token(user.id),
    )


@router.get("/me", response_model=UserResponse)
async def me(user: User = Depends(get_current_user)):
    return user

# backend/app/services/xml_parser.py
from lxml import etree


def parse_xml_bytes(raw: bytes) -> etree._Element:
    return etree.fromstring(raw)


def parse_xml_to_elements(raw: bytes) -> list[dict]:
    """Parse XML bytes into a flat list of element descriptors.

    Each element has: path, parent_path, level, element_name, value, is_leaf, attribute.
    Deduplicates by path — for repeated elements (like multiple <entry>),
    keeps the first occurrence's values as a sample.
    """
    root = parse_xml_bytes(raw)
    elements: list[dict] = []
    seen_paths: set[str] = set()

    def _local_name(tag: str) -> str:
        """Strip namespace URI but keep prefix if present in source."""
        if "}" in tag:
            return tag.split("}")[-1]
        return tag

    def _walk(el: etree._Element, parent_path: str, level: int):
        tag = _local_name(el.tag)
        # Reconstruct namespace prefix from nsmap
        for prefix, uri in el.nsmap.items():
            if uri and el.tag.startswith(f"{{{uri}}}"):
                tag = f"{prefix}:{_local_name(el.tag)}" if prefix else _local_name(el.tag)
                break

        path = f"{parent_path}/{tag}" if parent_path else tag
        has_children = len(el) > 0
        text = el.text.strip() if el.text and el.text.strip() else None

        if path not in seen_paths:
            seen_paths.add(path)
            elements.append({
                "path": path,
                "parent_path": parent_path or None,
                "level": level,
                "element_name": tag,
                "value": text,
                "is_leaf": not has_children and text is not None,
                "attribute": False,
            })

        # Process attributes
        for attr_name, attr_value in el.attrib.items():
            attr_path = f"{path}/@{attr_name}"
            if attr_path not in seen_paths:
                seen_paths.add(attr_path)
                elements.append({
                    "path": attr_path,
                    "parent_path": path,
                    "level": level + 1,
                    "element_name": f"@{attr_name}",
                    "value": attr_value,
                    "is_leaf": True,
                    "attribute": True,
                })

        for child in el:
            _walk(child, path, level + 1)

    _walk(root, "", 1)
    return elements
```

# backend/app/services/product_extractor.py
from lxml import etree


def _local_tag(el: etree._Element) -> str:
    tag = el.tag
    if "}" in tag:
        uri = tag.split("}")[0][1:]
        local = tag.split("}")[-1]
        for prefix, ns_uri in el.nsmap.items():
            if ns_uri == uri and prefix:
                return f"{prefix}:{local}"
        return local
    return tag


def _element_to_dict(el: etree._Element) -> dict:
    """Convert an XML element (a single product record) to a flat-ish dict."""
    result = {}

    # Include attributes with @ prefix
    for attr_name, attr_value in el.attrib.items():
        result[f"@{attr_name}"] = attr_value

    for child in el:
        tag = _local_tag(child)
        if len(child) > 0:
            # Nested element — recurse
            result[tag] = _element_to_dict(child)
        else:
            text = child.text.strip() if child.text and child.text.strip() else None
            result[tag] = text

    return result


def _find_record_elements(root: etree._Element, record_path: str) -> list[etree._Element]:
    """Find all elements matching the record_path in the XML tree."""
    parts = record_path.split("/")
    # The first part should match root
    root_tag = _local_tag(root)
    if root_tag != parts[0]:
        return []

    current = [root]
    for part in parts[1:]:
        next_level = []
        for el in current:
            for child in el:
                if _local_tag(child) == part:
                    next_level.append(child)
        current = next_level

    return current


def _get_product_name(record_el: etree._Element, product_name_path: str, record_path: str) -> str:
    """Extract product name from a record element using the relative path."""
    # product_name_path is absolute (e.g. "feed/entry/title")
    # We need the relative part after record_path (e.g. "title")
    relative = product_name_path
    if product_name_path.startswith(record_path + "/"):
        relative = product_name_path[len(record_path) + 1:]

    parts = relative.split("/")
    current = record_el
    for part in parts:
        found = None
        for child in current:
            if _local_tag(child) == part:
                found = child
                break
        if found is None:
            return ""
        current = found

    return current.text.strip() if current.text else ""


def extract_products(
    xml_bytes: bytes,
    record_path: str,
    product_name_path: str,
) -> list[dict]:
    """Extract products from XML. Returns list of {product_name, product_value}."""
    root = etree.fromstring(xml_bytes)
    record_elements = _find_record_elements(root, record_path)

    products = []
    for el in record_elements:
        product_value = _element_to_dict(el)
        product_name = _get_product_name(el, product_name_path, record_path)
        products.append({
            "product_name": product_name,
            "product_value": product_value,
        })

    return products
```

import re


def strip_currency(value: str | None) -> str | None:
    if value is None:
        return None
    return re.sub(r"\s*[A-Z]{3}\s*$", "", value.strip())


def format_price(value: str | None) -> str | None:
    if value is None:
        return None
    if value == "":
        return ""
    cleaned = strip_currency(value)
    try:
        return f"{float(cleaned):.2f}"
    except (ValueError, TypeError):
        return cleaned


def map_availability(value: str | None) -> str:
    if value is None:
        return "0"
    normalized = value.strip().lower()
    mapping = {
        "in stock": "1",
        "in_stock": "1",
        "available": "1",
        "out of stock": "0",
        "out_of_stock": "0",
        "preorder": "99",
        "pre-order": "99",
    }
    return mapping.get(normalized, "0")


def strip_html(value: str | None) -> str | None:
    if value is None:
        return None
    return re.sub(r"<[^>]+>", "", value).strip()
```

# backend/app/services/feed_generator.py
from lxml import etree

from app.transformers import format_price, map_availability, strip_html
from app.services.templates import CENEO_TEMPLATE

TRANSFORMS = {
    "format_price": format_price,
    "map_availability": map_availability,
    "strip_html": strip_html,
}


def _get_value(product_value: dict, source: str) -> str | None:
    """Get a value from product_value dict by source key."""
    return product_value.get(source)


def _apply_transform(value: str | None, transform_name: str | None) -> str | None:
    if transform_name is None or value is None:
        return value
    fn = TRANSFORMS.get(transform_name)
    if fn is None:
        return value
    return fn(value)


def generate_ceneo_xml(products: list[dict]) -> bytes:
    """Generate Ceneo XML from a list of product dicts (each has 'product_value' key)."""
    root = etree.Element("offers")

    for product in products:
        pv = product["product_value"]
        tmpl = CENEO_TEMPLATE

        o = etree.SubElement(root, "o")

        # Attributes on <o>
        id_val = _apply_transform(_get_value(pv, tmpl["id"]["source"]), tmpl["id"]["transform"])
        if id_val:
            o.set("id", id_val)

        url_val = _apply_transform(_get_value(pv, tmpl["url"]["source"]), tmpl["url"]["transform"])
        if url_val:
            o.set("url", url_val)

        price_val = _apply_transform(_get_value(pv, tmpl["price"]["source"]), tmpl["price"]["transform"])
        if price_val:
            o.set("price", price_val)

        avail_val = _apply_transform(_get_value(pv, tmpl["avail"]["source"]), tmpl["avail"]["transform"])
        o.set("avail", avail_val or "0")

        # Child elements
        cat_val = _apply_transform(_get_value(pv, tmpl["cat"]["source"]), tmpl["cat"]["transform"])
        if cat_val:
            cat_el = etree.SubElement(o, "cat")
            cat_el.text = cat_val

        name_val = _apply_transform(_get_value(pv, tmpl["name"]["source"]), tmpl["name"]["transform"])
        if name_val:
            name_el = etree.SubElement(o, "name")
            name_el.text = name_val

        desc_val = _apply_transform(_get_value(pv, tmpl["desc"]["source"]), tmpl["desc"]["transform"])
        if desc_val:
            desc_el = etree.SubElement(o, "desc")
            desc_el.text = desc_val

        # Images
        img_val = _apply_transform(_get_value(pv, tmpl["img"]["source"]), tmpl["img"]["transform"])
        if img_val:
            imgs_el = etree.SubElement(o, "imgs")
            main_el = etree.SubElement(imgs_el, "main")
            main_el.set("url", img_val)

        # Attrs
        attrs_el = etree.SubElement(o, "attrs")
        brand_val = _apply_transform(_get_value(pv, tmpl["brand"]["source"]), tmpl["brand"]["transform"])
        if brand_val:
            a = etree.SubElement(attrs_el, "a")
            a.set("name", "Producent")
            a.text = brand_val

        ean_val = _apply_transform(_get_value(pv, tmpl["ean"]["source"]), tmpl["ean"]["transform"])
        if ean_val:
            a = etree.SubElement(attrs_el, "a")
            a.set("name", "EAN")
            a.text = ean_val

        # Remove empty attrs
        if len(attrs_el) == 0:
            o.remove(attrs_el)

    return etree.tostring(root, encoding="UTF-8", xml_declaration=True, pretty_print=True)
```

from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.database import get_db
from app.deps import get_current_user
from app.models.user import User

    XmlElementResponse, ProductResponse,

router = APIRouter(prefix="/api/feeds-in", tags=["feeds-in"])


@router.get("", response_model=list[FeedInResponse])
async def list_feeds_in(
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
):
    result = await db.execute(
        select(FeedIn).where(FeedIn.user_id == user.id).order_by(FeedIn.created_at.desc())
    )
    return result.scalars().all()


@router.post("", response_model=FeedInResponse, status_code=status.HTTP_201_CREATED)
async def create_feed_in(
    body: FeedInCreate,
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
):
    feed = FeedIn(user_id=user.id, name=body.name, source_url=body.source_url)
    db.add(feed)
    await db.commit()
    await db.refresh(feed)
    return feed


@router.get("/{feed_id}", response_model=FeedInResponse)
async def get_feed_in(
    feed_id: int,
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
):
    feed = await _get_user_feed(db, feed_id, user.id)
    return feed


@router.put("/{feed_id}", response_model=FeedInResponse)
async def update_feed_in(
    feed_id: int,
    body: FeedInUpdate,
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
):
    feed = await _get_user_feed(db, feed_id, user.id)
    for field, value in body.model_dump(exclude_unset=True).items():
        setattr(feed, field, value)
    await db.commit()
    await db.refresh(feed)
    return feed


@router.delete("/{feed_id}", status_code=status.HTTP_204_NO_CONTENT)
async def delete_feed_in(
    feed_id: int,
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
):
    feed = await _get_user_feed(db, feed_id, user.id)
    await db.delete(feed)
    await db.commit()


@router.get("/{feed_id}/elements", response_model=list[XmlElementResponse])
async def get_elements(
    feed_id: int,
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
):
    await _get_user_feed(db, feed_id, user.id)
    result = await db.execute(
        select(XmlElementIn).where(XmlElementIn.feed_in_id == feed_id).order_by(XmlElementIn.id)
    )
    return result.scalars().all()


@router.get("/{feed_id}/products", response_model=list[ProductResponse])
async def get_products(
    feed_id: int,
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),
):
    await _get_user_feed(db, feed_id, user.id)
    result = await db.execute(
        select(ProductIn).where(ProductIn.feed_in_id == feed_id).order_by(ProductIn.id)
    )
    return result.scalars().all()


async def _get_user_feed(db: AsyncSession, feed_id: int, user_id: int) -> FeedIn:
    result = await db.execute(
        select(FeedIn).where(FeedIn.id == feed_id, FeedIn.user_id == user_id)
    )
    feed = result.scalar_one_or_none()
    if not feed:
        raise HTTPException(status_code=404, detail="Feed not found")
    return feed

# backend/app/tasks/feed_tasks.py
from datetime import datetime, timezone

import httpx
from sqlalchemy import create_engine
from sqlalchemy.orm import Session, sessionmaker

from app.celery_app import celery_app
from app.config import settings
from app.models.feed_in import FeedIn
from app.models.xml_element import XmlElementIn
from app.models.product import ProductIn
from app.services.xml_parser import parse_xml_to_elements
from app.services.product_extractor import extract_products


def fetch_xml_from_url(url: str) -> bytes:
    response = httpx.get(url, timeout=30, follow_redirects=True)
    response.raise_for_status()
    return response.content


def fetch_and_parse_sync(session: Session, feed_in_id: int) -> None:
    """Core sync logic for fetching and parsing a feed. Testable without Celery."""
    feed = session.query(FeedIn).get(feed_in_id)
    if not feed:
        return

    feed.fetch_status = "fetching"
    session.flush()

    try:
        xml_bytes = fetch_xml_from_url(feed.source_url)

        # Clear old data
        session.query(XmlElementIn).filter_by(feed_in_id=feed.id).delete()
        session.query(ProductIn).filter_by(feed_in_id=feed.id, custom_product=False).delete()

        # Parse elements
        elements = parse_xml_to_elements(xml_bytes)
        for el in elements:
            session.add(XmlElementIn(
                feed_in_id=feed.id,
                attribute=el["attribute"],
                path=el["path"],
                parent_path=el["parent_path"],
                level=el["level"],
                is_leaf=el["is_leaf"],
                element_name=el["element_name"],
                value=el["value"],
            ))

        # Extract products
        if feed.record_path and feed.product_name:
            products = extract_products(xml_bytes, feed.record_path, feed.product_name)
            for p in products:
                session.add(ProductIn(
                    feed_in_id=feed.id,
                    custom_product=False,
                    product_name=p["product_name"],
                    product_value=p["product_value"],
                ))

        feed.fetch_status = "success"
        feed.last_fetched_at = datetime.now(timezone.utc)
        session.commit()

    except Exception as e:
        session.rollback()
        feed.fetch_status = "error"
        session.commit()
        raise


@celery_app.task(name="feedy.fetch_and_parse")

    """Celery task wrapper."""
    engine = create_engine(settings.database_url_sync)
    session = sessionmaker(bind=engine)()
    try:
        fetch_and_parse_sync(session, feed_in_id)
        return {"status": "success", "feed_in_id": feed_in_id}
    finally:
        session.close()
        engine.dispose()

    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db),

    # Check feed_in belongs to user
    result = await db.execute(

        user_id=user.id,
        feed_in_id=body.feed_in_id,
        name=body.name,
        type=body.type,
        template=body.template,
        link_out=link_out,
    )

// frontend/src/stores/auth.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import api from '../api/client'

interface Plan {
  id: number
  name: string
  max_products: number | null
  max_feeds_out: number | null
  price_pln: number
}

interface User {
  id: number
  email: string
  plan: Plan
}

export const useAuthStore = defineStore('auth', () => {
  const user = ref<User | null>(null)
  const token = ref<string | null>(localStorage.getItem('access_token'))

  const isLoggedIn = computed(() => !!token.value)

  async function register(email: string, password: string) {
    const { data } = await api.post('/auth/register', { email, password })
    token.value = data.access_token
    localStorage.setItem('access_token', data.access_token)
    localStorage.setItem('refresh_token', data.refresh_token)
    await fetchUser()
  }

  async function login(email: string, password: string) {
    const { data } = await api.post('/auth/login', { email, password })
    token.value = data.access_token
    localStorage.setItem('access_token', data.access_token)
    localStorage.setItem('refresh_token', data.refresh_token)
    await fetchUser()
  }

  async function fetchUser() {
    const { data } = await api.get('/auth/me')
    user.value = data
  }

  function logout() {
    token.value = null
    user.value = null
    localStorage.removeItem('access_token')
    localStorage.removeItem('refresh_token')
  }

  return { user, token, isLoggedIn, register, login, fetchUser, logout }
})

  }

    const { data } = await api.post('/feeds-in', { name, source_url: sourceUrl })

    return data
  }

  async function updateFeed(id: number, updates: Partial<FeedIn>): Promise<FeedIn> {
    const { data } = await api.put(`/feeds-in/${id}`, updates)
    const idx = feeds.value.findIndex(f => f.id === id)
    if (idx !== -1) feeds.value[idx] = data
    return data
  }

  async function deleteFeed(id: number) {
    await api.delete(`/feeds-in/${id}`)
    feeds.value = feeds.value.filter(f => f.id !== id)
  }

  async function fetchFeedXml(id: number) {
    await api.post(`/feeds-in/${id}/fetch`)
  }

  async function getElements(id: number): Promise<XmlElement[]> {
    const { data } = await api.get(`/feeds-in/${id}/elements`)
    return data
  }

  async function getProducts(id: number): Promise<Product[]> {
    const { data } = await api.get(`/feeds-in/${id}/products`)
    return data
  }

  return { feeds, fetchFeeds, createFeed, updateFeed, deleteFeed, fetchFeedXml, getElements, getProducts }

  { value: 'ceneo', label: 'Ceneo', desc: 'Gotowe mapowanie pól' },

  { value: 'custom', label: 'Custom', desc: 'Ręczne mapowanie' },
]

async function handleCreate() {
  loading.value = true
  error.value = ''
  try {
    const feedOut = await feedsOut.createFeed({
      feed_in_id: feedInId,
      name: name.value,
      type: selectedTemplate.value,
      template: selectedTemplate.value,
    })

  } catch (e: any) {
    error.value = e.response?.data?.detail || 'Błąd tworzenia feeda'
  } finally {
    loading.value = false
  }
}

<template>
  <div class="max-w-5xl mx-auto px-4 py-8">
    <div v-if="!feedOut" class="text-center py-16 text-gray-400">Ładowanie...</div>
    <template v-else>

      </div>

      <!-- Mapping Table -->
      <div class="bg-white rounded-lg shadow-sm border p-4 mb-6">
        <h2 class="font-semibold mb-3">Mapowanie pól</h2>
        <MappingTable :rows="structure" :sample-product="sampleProduct" @remove="removeRow" @add="addRow" />
        <div class="mt-4 flex gap-3">
          <button @click="saveStructure"
            class="bg-indigo-600 text-white px-4 py-2 rounded-lg hover:bg-indigo-700 text-sm">
            Zapisz mapowanie
          </button>
        </div>
      </div>

      <!-- Generated Link -->
      <div class="bg-white rounded-lg shadow-sm border p-4">
        <h2 class="font-semibold mb-3">Link do feeda</h2>
        <div class="flex items-center gap-3">
          <code class="bg-gray-100 px-3 py-2 rounded text-sm flex-1 truncate">
            {{ feedUrl }}
          </code>
          <button @click="copyLink" class="bg-gray-100 px-3 py-2 rounded text-sm hover:bg-gray-200">
            {{ copied ? 'Skopiowano!' : 'Kopiuj' }}
          </button>
          <a :href="feedUrl" target="_blank" class="text-indigo-600 text-sm hover:underline">Podgląd XML</a>
        </div>
      </div>
    </template>
  </div>

<script setup lang="ts">
import { ref, computed, onMounted } from 'vue'
import { useRoute } from 'vue-router'
import { useFeedsOutStore, type FeedOut, type StructureElement } from '../stores/feedsOut'
import { useFeedsInStore } from '../stores/feedsIn'
import MappingTable from '../components/MappingTable.vue'

const route = useRoute()
const feedsOut = useFeedsOutStore()
const feedsIn = useFeedsInStore()

const feedOutId = Number(route.params.id)
const feedOut = ref<FeedOut | null>(null)
const structure = ref<StructureElement[]>([])
const sampleProduct = ref<Record<string, any> | null>(null)
const copied = ref(false)

const feedUrl = computed(() =>
  feedOut.value ? `${window.location.origin}/feed/${feedOut.value.link_out}.xml` : ''
)

onMounted(async () => {
  await feedsOut.fetchFeeds()
  feedOut.value = feedsOut.feeds.find(f => f.id === feedOutId) || null
  if (feedOut.value) {
    structure.value = await feedsOut.getStructure(feedOutId)
    const products = await feedsIn.getProducts(feedOut.value.feed_in_id)
    if (products.length > 0) {
      sampleProduct.value = products[0].product_value
    }
  }
})

function addRow() {
  const idx = structure.value.length + 1
  structure.value.push({
    sort_key: String(idx),
    custom_element: false,
    path_in: null,
    level_out: 2,
    path_out: '',
    parent_path_out: null,
    element_name_out: '',
    is_leaf: false,
    attribute: false,
  })
}

function removeRow(index: number) {
  structure.value.splice(index, 1)
}

async function saveStructure() {
  structure.value = await feedsOut.updateStructure(feedOutId, structure.value)
}

async function copyLink() {
  await navigator.clipboard.writeText(feedUrl.value)
  copied.value = true
  setTimeout(() => copied.value = false, 2000)
}
</script>
